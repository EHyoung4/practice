import turtle as t
import random
t.shape("turtle")

def draw_snowman(x,y):
    t.speed(0)
    t.up();t.goto(x,y+140);t.down()
    t.begin_fill();t.color("black","white");t.circle(40);t.end_fill()
    t.up();t.goto(x,y+100);t.down()
    t.begin_fill();t.color("black","white");t.circle(30);t.end_fill()
    t.up();t.goto(x,y);t.down()
    t.begin_fill();t.color("black","white");t.circle(60);t.end_fill()
    t.up();t.goto(x+30,y+100);t.down()
    t.lt(25);t.fd(50);
    t.up();t.goto(x-80,y+100);t.down()
    t.rt(25);t.fd(50);

s=t.Screen();s.bgcolor('skyblue');
for i in range(3):
    x=random.randrange(-300,300)
    y=random.randrange(-100,100)
    draw_snowman(x,y)
