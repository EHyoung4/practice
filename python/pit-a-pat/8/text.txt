import turtle as t
def draw_shape(r,c1):
    t.lt(270);t.width(3);t.color("black",c1)
    t.begin_fill()
    t.circle(r/2.0, -180);t.circle(r,180)
    t.lt(180);t.circle(-r/2.0,-180)
    t.end_fill()

t.reset()
draw_shape(200,"red")
t.setheading(180)
draw_shape(200,"blue")
