import turtle as t
import random
color=["blue","red","purple","yellow","orange","pink"]

def draw(x,y,c,n):
    t.up();t.goto(x,y);t.down();t.speed(0);
    t.begin_fill()
    for i in range(n):
        t.fd(50);t.lt(360/n);
    t.color("black", c);t.end_fill()

for i in range(10):
    x=random.randrange(-200,200)
    y=random.randrange(-200,200)
    c=random.choice(color)
    n=random.randrange(3,8)
    draw(x,y,c,n)

import turtle as t
import random
color=["blue","red","purple","yellow","orange","pink"]

def draw_star(c,length,x,y):
    t.up();t.goto(x,y);t.down();t.speed(0);
    t.begin_fill()
    i=0
    while i<5:
        t.fd(length);t.rt(144);i+=1
    t.color(c,c);t.end_fill()
    
s=t.Screen();s.bgcolor("black")
for i in range(10):
    x=random.randrange(-200,200)
    y=random.randrange(-200,200)
    c=random.choice(color)
    length=random.randrange(0,100)
    draw_star(c,length,x,y)
